using System;

namespace ForLoopBreakContinue
{
    class MainClass
    {
        public static void Main(string[] args)
        {


            int n = 2;

            for (int i=1; i<=10; i++)
            {

                Console.WriteLine(n + " x " + i + " = " + i * n);

            }




        }
    }
}
